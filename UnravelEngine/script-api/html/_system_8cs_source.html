141 // The core logic for iteration in priority order, plus deferred add & remove cleanup
142 private void InvokeInternal(Action<ScriptComponent> action)
143 {
144 isInvoking = true;
145 try
146 {
147 // Iterate in current sorted order
148 foreach (var entry in entries)
149 {
150 if (entry.Comp != null && !IsRemoved(entry.Comp))
151 {
152 action(entry.Comp);
153 }
154 }
155 }
156 finally
157 {
158 isInvoking = false;
159 }
160
161 // Insert any pending additions
162 if (pendingOps.Count > 0)
163 {
164 foreach (var op in pendingOps)
165 {
166 if (op.add)
167 {
168 InsertComponent(op.Comp);
169 }
170 else
171 {
172 RemoveComponent(op.Comp);
173 }
174 }
175 pendingOps.Clear();
176
177 // Re-sort
178 Resort();
179 }
180 }

233 private void CollapseOperations(ScriptComponent comp, bool isAdd)
234 {
235 // Find the last operation for this component
236 int lastOpIndex = -1;
237 for (int i = pendingOps.Count - 1; i >= 0; i--)
238 {
239 if (pendingOps[i].Comp == comp)
240 {
241 lastOpIndex = i;
242 break;
243 }
244 }
245
246 if (lastOpIndex == -1)
247 {
248 // No previous operations for this component, just add the new one
249 pendingOps.Add(new PendingOp { add = isAdd, Comp = comp });
250 }
251 else
252 {
253 var lastOp = pendingOps[lastOpIndex];
254
255 if (lastOp.add && !isAdd)
256 {
257 // Last was add, this is remove -> cancel both operations
258 pendingOps.RemoveAt(lastOpIndex);
259 }
260 else if (!lastOp.add && isAdd)
261 {
262 // Last was remove, this is add -> cancel both operations
263 pendingOps.RemoveAt(lastOpIndex);
264 }
265 else if (lastOp.add && isAdd)
266 {
267 // Last was add, this is add -> do nothing (already added)
268 // No need to add another add operation
269 }
270 else if (!lastOp.add && !isAdd)
271 {
272 // Last was remove, this is remove -> do nothing (already removed)
273 // No need to add another remove operation
274 }
275 }
