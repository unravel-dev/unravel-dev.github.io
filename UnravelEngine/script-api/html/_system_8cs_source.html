156 // The core logic for iteration in priority order, plus deferred add & remove cleanup
157 private void InvokeInternal(Action<ScriptComponent> action)
158 {
159 isInvoking = true;
160 removedDuringInvoke.Clear(); // Clear at start of invocation
161
162 try
163 {
164 // Iterate in current sorted order
165 foreach (var entry in entries)
166 {
167 if (!ReferenceEquals(entry.Comp, null) && !removedDuringInvoke.Contains(entry.Comp))
168 // if (entry.Comp != null && !removedDuringInvoke.Contains(entry.Comp))
169 {
170 action(entry.Comp);
171 }
172 }
173 }
174 finally
175 {
176 isInvoking = false;
177 }
178
179 // Insert any pending additions
180 if (pendingOps.Count > 0)
181 {
182 foreach (var op in pendingOps)
183 {
184 if (op.add)
185 {
186 InsertComponent(op.Comp);
187 }
188 else
189 {
190 RemoveComponent(op.Comp);
191 }
192 }
193 pendingOps.Clear();
194
195 // Re-sort
196 Resort();
197 }

253 private void CollapseOperations(ScriptComponent comp, bool isAdd)
254 {
255 // Find the last operation for this component
256 int lastOpIndex = -1;
257 for (int i = pendingOps.Count - 1; i >= 0; i--)
258 {
259 if (pendingOps[i].Comp == comp)
260 {
261 lastOpIndex = i;
262 break;
263 }
264 }
265
266 if (lastOpIndex == -1)
267 {
268 // No previous operations for this component, just add the new one
269 pendingOps.Add(new PendingOp { add = isAdd, Comp = comp });
270
271 // Track removal for fast lookup during invoke
272 if (!isAdd)
273 {
274 removedDuringInvoke.Add(comp);
275 }
276 }
277 else
278 {
279 var lastOp = pendingOps[lastOpIndex];
280
281 if (lastOp.add && !isAdd)
282 {
283 // Last was add, this is remove -> cancel both operations
284 pendingOps.RemoveAt(lastOpIndex);
285 removedDuringInvoke.Add(comp);
286 }
287 else if (!lastOp.add && isAdd)
288 {
289 // Last was remove, this is add -> cancel both operations
290 pendingOps.RemoveAt(lastOpIndex);
291 removedDuringInvoke.Remove(comp);
292 }
293 else if (lastOp.add && isAdd)
294 {
295 // Last was add, this is add -> do nothing (already added)
296 // No need to add another add operation
297 }
298 else if (!lastOp.add && !isAdd)
299 {
300 // Last was remove, this is remove -> do nothing (already removed)
301 // No need to add another remove operation
302 }
303 }

341 public void CheckAndLog(string context = "")
342 {
343 int gen0 = GC.CollectionCount(0);
344 int gen1 = GC.CollectionCount(1);
345 int gen2 = GC.CollectionCount(2);
346 long memory = GC.GetTotalMemory(false);
347
348 int gen0Delta = gen0 - lastGen0Count;
349 int gen1Delta = gen1 - lastGen1Count;
350 int gen2Delta = gen2 - lastGen2Count;
351 long memDelta = memory - lastMemory;
352
353 // Log every time (even if no change) for debugging
354 if (gen0Delta > 0 || gen1Delta > 0 || gen2Delta > 0 || Math.Abs(memDelta) > 1024)
355 {
356 // Get Mono heap size (native memory used by Mono runtime)
357 long monoHeap = internal_m2n_get_mono_heap_size();
358 long monoUsed = internal_m2n_get_mono_used_size();
359
360 Log.Info($"[GC] {context} - Collections: Gen0={gen0Delta}, Gen1={gen1Delta}, Gen2={gen2Delta} | Managed: {memDelta / 1024.0:F2} KB (Total: {memory / 1024.0:F2} KB) | Mono Heap: {monoHeap / (1024.0 * 1024.0):F2} MB (Used: {monoUsed / (1024.0 * 1024.0):F2} MB)");
361 }
362
363 lastGen0Count = gen0;
364 lastGen1Count = gen1;
365 lastGen2Count = gen2;
366 lastMemory = memory;
