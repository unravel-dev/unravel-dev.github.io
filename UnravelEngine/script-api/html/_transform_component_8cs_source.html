492 public void RotateAround(Vector3 point, Vector3 axis, float angle)
493 {
494 Vector3 vector = position;
495 Quaternion quaternion = Quaternion.AngleAxis(angle, axis);
496 Vector3 vector2 = vector - point;
497 vector2 = quaternion * vector2;
498 vector = point + vector2;
499 position = vector;
500 RotateAxis(angle, axis);
501 }
