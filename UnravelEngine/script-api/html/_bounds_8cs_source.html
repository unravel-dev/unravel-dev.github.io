177 // helper to check one axis
178 bool Slab(float o, float d, float mn, float mx, ref float t0, ref float t1)
179 {
180 if (MathF.Abs(d) > float.Epsilon)
181 {
182 float inv = 1f / d;
183 float t1n = (mn - o) * inv;
184 float t2n = (mx - o) * inv;
185 if (t1n > t2n) (t1n, t2n) = (t2n, t1n);
186 t0 = MathF.Max(t0, t1n);
187 t1 = MathF.Min(t1, t2n);
188 return t0 <= t1;
189 }
190 else
191 {
192 return o >= mn && o <= mx;
193 }
194 }

200 public bool Intersect(Vector3 origin, Vector3 dir, out float t, bool restrictRange = true)
201 {
202 t = 0f;
203 float tMin = float.MinValue;
204 float tMax = float.MaxValue;
205
206 if (!Slab(origin.x, dir.x, Min.x, Max.x, ref tMin, ref tMax)) return false;
207 if (!Slab(origin.y, dir.y, Min.y, Max.y, ref tMin, ref tMax)) return false;
208 if (!Slab(origin.z, dir.z, Min.z, Max.z, ref tMin, ref tMax)) return false;
209
210 t = tMin > 0 ? tMin : tMax;
211 return t >= 0 && (!restrictRange || t <= 1f);
212 }
