21 public static void StableSort<T, TKey>(
22 this IList<T> list,
23 Func<T, TKey> keySelector,
24 IComparer<TKey> keyComparer = null,
25 bool usePooling = false)
26 {
27 if (list == null) throw new ArgumentNullException(nameof(list));
28 if (keySelector == null) throw new ArgumentNullException(nameof(keySelector));
29
30 if (keyComparer == null)
31 keyComparer = Comparer<TKey>.Default;
32
33 // We store "keys" in a parallel array so we only compute keySelector once per element
34 int count = list.Count;
35 if (count <= 1) return; // Nothing to sort
36
37 // Get or create key array (pooled or allocated)
38 TKey[] keys;
39 if (usePooling)
40 {
41 keys = GetOrCreateKeyArray<TKey>(count);
42 }
43 else
44 {
45 keys = new TKey[count];
46 }
47
48 for (int i = 0; i < count; i++)
49 {
50 keys[i] = keySelector(list[i]);
51 }
52
53 // mergesort
54 MergesortInPlace(list, keys, 0, count, keyComparer, usePooling);
55
56 // Clear pooled key array to avoid keeping references alive
57 if (usePooling)
58 {
59 Array.Clear(keys, 0, count);
60 }
61 }

63 private static TKey[] GetOrCreateKeyArray<TKey>(int minSize)
64 {
65 if (keyArrayPool == null)
66 keyArrayPool = new Dictionary<Type, object>();
67
68 var keyType = typeof(TKey);
69
70 if (!keyArrayPool.TryGetValue(keyType, out object pooledObj) || pooledObj == null)
71 {
72 var newArray = new TKey[Math.Max(minSize, 64)]; // Start with reasonable size
73 keyArrayPool[keyType] = newArray;
74 return newArray;
75 }
76
77 var pooled = (TKey[])pooledObj;
78 if (pooled.Length < minSize)
79 {
80 // Need to grow
81 var newArray = new TKey[Math.Max(minSize, pooled.Length * 2)];
82 keyArrayPool[keyType] = newArray;
83 return newArray;
84 }
85
86 return pooled;

115 // Recursively split [start..end) into two halves, then merge them stably
116 private static void MergesortInPlace<T, TKey>(
117 IList<T> list,
118 TKey[] keys,
119 int start,
120 int end,
121 IComparer<TKey> comparer,
122 bool usePooling)
123 {
124 int length = end - start;
125 if (length <= 1) return;
126
127 int mid = (start + end) / 2;
128 MergesortInPlace(list, keys, start, mid, comparer, usePooling);
129 MergesortInPlace(list, keys, mid, end, comparer, usePooling);
130
131 // Merge two sorted halves: [start..mid), [mid..end)
132 Merge(list, keys, start, mid, end, comparer, usePooling);
133 }

135 // Merge two sorted subranges in place, stably.
136 private static void Merge<T, TKey>(
137 IList<T> list,
138 TKey[] keys,
139 int start,
140 int mid,
141 int end,
142 IComparer<TKey> comparer,
143 bool usePooling)
144 {
145 int leftSize = mid - start;
146 int rightSize = end - mid;
147
148 // We copy the left half into a temp array (pooled or allocated based on usePooling)
149 // (the right half stays in place, and we merge them in 'list' itself)
150 T[] leftElements;
151 TKey[] leftKeys;
152
153 if (usePooling)
154 {
155 leftElements = GetOrCreateElementArray<T>(leftSize);
156 leftKeys = GetOrCreateKeyArray<TKey>(leftSize);
157 }
158 else
159 {
160 leftElements = new T[leftSize];
161 leftKeys = new TKey[leftSize];
162 }
163
164 for (int i = 0; i < leftSize; i++)
165 {
166 leftElements[i] = list[start + i];
167 leftKeys[i] = keys[start + i];
168 }
169
170 // We'll merge them back into [start..end) of 'list'
171 int leftIndex = 0; // index into left half
172 int rightIndex = mid; // index into 'list' for the right half
173 int destIndex = start;
174
175 while (leftIndex < leftSize && rightIndex < end)
176 {
177 // Compare left vs. right
178 int cmp = comparer.Compare(leftKeys[leftIndex], keys[rightIndex]);
179 // If left <= right, we take left to maintain stability (i.e. we do <=, not <)
180 if (cmp <= 0)
181 {
182 list[destIndex] = leftElements[leftIndex];
183 keys[destIndex] = leftKeys[leftIndex];
184 leftIndex++;
185 }
186 else
187 {
188 list[destIndex] = list[rightIndex];
189 keys[destIndex] = keys[rightIndex];
190 rightIndex++;
191 }
192 destIndex++;
193 }
194
195 // Copy any remaining left side
196 while (leftIndex < leftSize)
197 {
198 list[destIndex] = leftElements[leftIndex];
199 keys[destIndex] = leftKeys[leftIndex];
200 leftIndex++;
201 destIndex++;
202 }
203
204 // Right side is already in place if any remain, so no extra loop needed
205
206 // Clear pooled arrays to avoid keeping references alive
207 if (usePooling)
208 {
209 Array.Clear(leftElements, 0, leftSize);
210 Array.Clear(leftKeys, 0, leftSize);
211 }
212 }
