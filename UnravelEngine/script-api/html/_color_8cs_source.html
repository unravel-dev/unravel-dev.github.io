458 private static void RGBToHSVHelper(float offset, float dominantcolor, float colorone, float colortwo, out float H, out float S, out float V)
459 {
460 V = dominantcolor;
461 if (V != 0f)
462 {
463 float num = 0f;
464 num = ((!(colorone > colortwo)) ? colorone : colortwo);
465 float num2 = V - num;
466 if (num2 != 0f)
467 {
468 S = num2 / V;
469 H = offset + (colorone - colortwo) / num2;
470 }
471 else
472 {
473 S = 0f;

480 H += 1f;

531 //
532 //
533 // Returns:
534 // An opaque colour with HSV matching the input.
535 public static Color HSVToRGB(float H, float S, float V, bool hdr)
536 {
537 Color result = white;
538 if (S == 0f)
539 {
540 result.r = V;
541 result.g = V;
542 result.b = V;
543 }
544 else if (V == 0f)
545 {
546 result.r = 0f;
547 result.g = 0f;
548 result.b = 0f;
549 }
550 else
551 {
552 result.r = 0f;
553 result.g = 0f;
554 result.b = 0f;
555 float num = H * 6f;
556 int num2 = (int)Mathf.Floor(num);
557 float num3 = num - (float)num2;
558 float num4 = V * (1f - S);
559 float num5 = V * (1f - S * num3);
560 float num6 = V * (1f - S * (1f - num3));
561 switch (num2)
562 {
563 case 0:
564 result.r = V;
565 result.g = num6;
566 result.b = num4;

569 result.r = num5;
570 result.g = V;

574 result.r = num4;

577 break;
578 case 3:
579 result.r = num4;
580 result.g = num5;
581 result.b = V;
582 break;
583 case 4:
584 result.r = num6;

587 break;

592 break;
