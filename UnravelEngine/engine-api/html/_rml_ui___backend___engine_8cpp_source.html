40auto initialize(rtti::context& ctx, const char* window_name, int width, int height) -> bool
41{
42 APPLOG_TRACE("{}::{}", "RmlUi_Backend_Engine", __func__);
43
44 if (data)
45 {
46 APPLOG_ERROR("RmlUi backend already initialized");
47 return false;
48 }
49
50 data = std::make_unique<BackendData>();
51 data->engine_ctx = &ctx;
52
53 // Initialize system interface
54 if (!data->system_interface.init(ctx))
55 {
56 APPLOG_ERROR("Failed to initialize RmlUi system interface");
57 data.reset();
58 return false;
59 }
60
61 // Initialize render interface
62 if (!data->render_interface.init(ctx))
63 {
64 APPLOG_ERROR("Failed to initialize RmlUi render interface");
65 data->system_interface.shutdown();
66 data.reset();
67 return false;
68 }
69
70 // Set up RmlUi with our interfaces
71 Rml::SetSystemInterface(&data->system_interface);
72 Rml::SetRenderInterface(&data->render_interface);
73 Rml::SetFileInterface(&data->file_interface);
74
75 // Initialize RmlUi core
76 if (!Rml::Initialise())
77 {
78 APPLOG_ERROR("Failed to initialize RmlUi core");
79 data->render_interface.shutdown();
80 data->system_interface.shutdown();
81 data.reset();
82 return false;
83 }
84
85 // Set viewport size
86 data->render_interface.set_viewport(width, height);
87
88
89 data->initialized = true;
90 return true;
91}

93void shutdown()
94{
95 APPLOG_TRACE("{}::{}", "RmlUi_Backend_Engine", __func__);
96
97 if (!data)

99 APPLOG_WARNING("RmlUi backend not initialized");

103 if (data->initialized)

105 // Shutdown RmlUi core

107
108 // Cleanup our interfaces
109 data->render_interface.shutdown();
110 data->system_interface.shutdown();
111
112 data->initialized = false;
113 }
114
115 data.reset();
