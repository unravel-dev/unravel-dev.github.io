70void RmlUi_SystemInterface::SetMouseCursor(const Rml::String& cursor_name)
71{
72 auto& ctx = engine::context();
73 auto window = ctx.get_cached<renderer>().get_main_window();
74 if (!window)
75 {
76 return;
77 }
78
79 // Map RmlUi cursor names to engine cursor types
80 os::cursor::type cursor_type = os::cursor::type::arrow;
81
82 if (cursor_name == "pointer" || cursor_name == "hand")
83 {
84 cursor_type = os::cursor::type::hand;
85 }
86 else if (cursor_name == "text" || cursor_name == "caret")
87 {
88 cursor_type = os::cursor::type::ibeam;
89 }
90 else if (cursor_name == "cross")
91 {
92 cursor_type = os::cursor::type::crosshair;
93 }
94 else if (cursor_name == "move")
95 {
96 cursor_type = os::cursor::type::size_all;
97 }
98 else if (cursor_name.find("resize") != Rml::String::npos)
99 {
100 // Handle different resize cursors
101 if (cursor_name.find("n-resize") != Rml::String::npos ||
102 cursor_name.find("s-resize") != Rml::String::npos)
103 {
104 cursor_type = os::cursor::type::size_ns;
105 }
106 else if (cursor_name.find("e-resize") != Rml::String::npos ||
107 cursor_name.find("w-resize") != Rml::String::npos)
108 {
109 cursor_type = os::cursor::type::size_we;
110 }
111 else if (cursor_name.find("ne-resize") != Rml::String::npos ||
112 cursor_name.find("sw-resize") != Rml::String::npos)
113 {
114 cursor_type = os::cursor::type::size_nesw;
115 }
116 else if (cursor_name.find("nw-resize") != Rml::String::npos ||
117 cursor_name.find("se-resize") != Rml::String::npos)
118 {
119 cursor_type = os::cursor::type::size_nwse;
120 }
121 }
122 else if (cursor_name == "unavailable" || cursor_name == "not-allowed")
123 {
124 cursor_type = os::cursor::type::not_allowed;
125 }
126
127 // Set cursor on the window
128 window->get_window().set_cursor(cursor_type);
129}

197auto input_event_handler(Rml::Context* context, const os::event& event) -> bool
198{
199 if (!context)
200 {
201 return true; // Event continues propagating
202 }
203
204 bool handled = false;
205
206 switch (event.type)
207 {
208 case os::events::key_down:
209 {
210 auto rml_key = convert_key(event.key.code);
211 auto modifiers = get_key_modifier_state();
212 handled = context->ProcessKeyDown(rml_key, modifiers);
213 if (event.key.code == os::key::code::enter || event.key.code == os::key::code::kp_enter)
214 {
215 handled |= context->ProcessTextInput('\n');
216 }
217 break;
218 }
219
220 case os::events::key_up:
221 {
222 auto rml_key = convert_key(event.key.code);
223 auto modifiers = get_key_modifier_state();
224 handled = context->ProcessKeyUp(rml_key, modifiers);
225 break;
226 }
227
228 case os::events::text_input:
229 {
230 // Convert text input to RmlUi character events
231 for (char c : event.text.text)
232 {
233 if (c != 0)
234 {
235 handled = context->ProcessTextInput(c) || handled;
236 }
237 }
238 break;
239 }
240
241 case os::events::mouse_button:
242 {
243 auto rml_button = convert_mouse_button(event.button.button);
244 auto modifiers = get_key_modifier_state();
245
246 if (event.button.state_id == os::state::pressed)
247 {
248 handled = context->ProcessMouseButtonDown(rml_button, modifiers);
249 }
250 else if (event.button.state_id == os::state::released)
251 {
252 handled = context->ProcessMouseButtonUp(rml_button, modifiers);
253 }
254 break;
255 }
256
257 // case os::events::mouse_motion:
258 // {
259 // auto modifiers = get_key_modifier_state();
260 // handled = context->ProcessMouseMove(event.motion.x, event.motion.y, modifiers);
261 // break;
262 // }
263
264 case os::events::mouse_wheel:
265 {
266 auto modifiers = get_key_modifier_state();
267 // RmlUi expects wheel delta as integer, ospp provides float
268 float wheel_delta = static_cast<float>(event.wheel.y);
269 handled = context->ProcessMouseWheel(-wheel_delta, modifiers);
270 break;
271 }
272
273 default:
274 // Event not handled by RmlUi
275 break;
276 }
277
278 return !handled; // Return true if event should continue propagating
279}
