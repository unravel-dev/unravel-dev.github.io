46uint32_t utf8_decode(uint32_t* _state, uint32_t* _codep, uint8_t _ch)
47{
48 uint32_t byte = _ch;
49 uint32_t type = s_utf8d[byte];
50
51 *_codep = (*_state != UTF8_ACCEPT) ?
52 (byte & 0x3fu) | (*_codep << 6) :
53 (0xff >> type) & (byte);
54
55 *_state = s_utf8d[256 + *_state + type];
56 return *_state;
57}
