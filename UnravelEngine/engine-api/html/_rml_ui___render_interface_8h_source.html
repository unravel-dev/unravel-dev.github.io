53template<typename CompiledType, typename InternalHandle, typename RmlHandle, size_t MaxHandles>
54class compiled_resource_manager
55{
56public:
57 compiled_resource_manager() = default;
58 ~compiled_resource_manager() = default;
59
60 // Non-copyable, non-movable for safety
61 compiled_resource_manager(const compiled_resource_manager&) = delete;
62 compiled_resource_manager& operator=(const compiled_resource_manager&) = delete;
63 compiled_resource_manager(compiled_resource_manager&&) = delete;
64 compiled_resource_manager& operator=(compiled_resource_manager&&) = delete;
65
70 auto alloc() -> uint16_t
71 {
72 return handle_allocator_.alloc();
73 }
74
79 void free(uint16_t handle_idx)
80 {
81 if(handle_allocator_.isValid(handle_idx))
82 {
83 // Clear the resource entry
84 compiled_resources_[handle_idx] = CompiledType{};
85 handle_allocator_.free(handle_idx);
86 }
87 }
88
94 auto is_valid(uint16_t handle_idx) const -> bool
95 {
96 return handle_allocator_.isValid(handle_idx);
97 }
98
104 auto get(uint16_t handle_idx) -> CompiledType&
105 {
106 return compiled_resources_[handle_idx];
107 }
108
114 auto get(uint16_t handle_idx) const -> const CompiledType&
115 {
116 return compiled_resources_[handle_idx];
117 }
118
124 static auto to_rml_handle(InternalHandle handle) -> RmlHandle
125 {
126 return static_cast<RmlHandle>(handle.idx + 1);
127 }
128
134 static auto from_rml_handle(RmlHandle handle) -> InternalHandle
135 {
136 InternalHandle result;
137 result.idx = (handle > 0) ? static_cast<uint16_t>(handle - 1) : bx::kInvalidHandle;
138 return result;
139 }
140
145 auto get_num_handles() const -> uint16_t
146 {
147 return handle_allocator_.getNumHandles();
148 }
149
154 template<typename CleanupFunc>
155 void cleanup_all(CleanupFunc cleanup_func)
156 {
157 auto handles_copy = handle_allocator_;
158 for(uint16_t i = 0; i < handles_copy.getNumHandles(); ++i)
159 {
160 if(handles_copy.isValid(i))
161 {
162 cleanup_func(compiled_resources_[i]);
163 free(i);
164 }
165 }
166 }
167
168private:
169 bx::HandleAllocT<MaxHandles> handle_allocator_;
170 std::array<CompiledType, MaxHandles> compiled_resources_;
171};
